How to calculate two text fields values in javascript
=============================================

<!DOCTYPE html>
<html>
	<head>		
	</head>
	<body>
		<h1>Calculate Two fields</h1>
		
		<form>
		
			No1:<input type="text" id="t1"/><br>
			No2:<input type="text" id="t2"/><br>
			<input type="button" value="calculate" onclick="calcu()"/>	
			
			<h1 id="answer"></h1>
		</form>

		<script>
			function calcu()
			{
				var val1=document.getElementById("t1").value;
				var val2=document.getElementById("t2").value;
				var result=parseFloat(val1)+parseFloat(val2);
			
				if(!isNaN(result))
				{
			document.getElementById("answer").innerHTML="The Result is :"+result;
				}
			}   
    </script>  
		
	</body>
	
</html>

JavaScript Hide and Show a portion of a form
========================================

<!DOCTYPE html>
<html>
	<head>
			<script>
					function hide()
					{
							if(document.getElementById('box').checked)
							{
								document.getElementById('shopinfo').style.display='none';
							}
							else
							{
								document.getElementById('shopinfo').style.display='block';
							}
					}

			</script>
			
	</head>
	<body>
			<fieldset id="billinfo">
			<legend>Address</legend>
			No:<input type="text" id="t1"/><br>
			Name:<input type="text" id="t2"/><br>
			Address:<input type="text" id="t3"/><br>
